# -- PROTEIN FUNCTION ANALYSIS PROJECT -- 

# BASIC SET UP
CMAKE_MINIMUM_REQUIRED( VERSION 2.6 )
PROJECT( ProtFun )

# DEFINE VERSION VARIABLES
SET( NETNGLYC_VERSION 1.0 )
SET( NETOGLYC_VERSION 3.1d )
SET( PROP_VERSION 1.0c)
SET( PROTPARAM_VERSION 1.0 )
SET( PSIPRED_VERSION 2.5 )
SET( SEG_VERSION 1 )
SET( SIGNALP_VERSION 4.1 )
SET( TMHMM_VERSION 2.0c )

# DEFINE SUB PROJECT DIRECTORIES
SET( NETNGLYC_DIR netNglyc-${NETNGLYC_VERSION} )
SET( NETOGLYC_DIR netOglyc-${NETOGLYC_VERSION} )
SET( PROP_DIR prop-${PROP_VERSION} )
SET( PROTPARAM_DIR protparam-${PROTPARAM_VERSION} )
SET( PSIPRED_DIR psipred-${PSIPRED_VERSION} )
SET( SEG_DIR seg-${SEG_VERSION} )
SET( SIGNALP_DIR signalp-${SIGNALP_VERSION} )
SET( TMHMM_DIR tmhmm-${TMHMM_VERSION} )

# DEFINE INSTALLATION DIRECTORIES
SET( BIN_DIR /usr/biotools/ )
SET( DATA_DIR /usr/share/ )
SET( TEMP_DIR /var/tmp/ )
SET( MAN_DIR /usr/share/man/man1/ )

# ADD SUB PROJECT DIRECTORIES
ADD_SUBDIRECTORY( ${SIGNALP_DIR} )
ADD_SUBDIRECTORY( ${PSIPRED_DIR} )
ADD_SUBDIRECTORY( ${NETNGLYC_DIR} )
ADD_SUBDIRECTORY( ${NETOGLYC_DIR} )
ADD_SUBDIRECTORY( ${PROP_DIR} )
ADD_SUBDIRECTORY( ${PROTPARAM_DIR} )
ADD_SUBDIRECTORY( ${TMHMM_DIR} )
